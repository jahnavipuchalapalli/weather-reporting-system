% Simulated weather data
num_samples = 100; % Number of samples
temperature_data = randi([20, 35], 1, num_samples); % Simulated temperature data (in Celsius)
humidity_data = randi([40, 80], 1, num_samples); % Simulated humidity data (in percentage)
rainfall_data = randi([0, 50], 1, num_samples); % Simulated rainfall data (in mm)
 
% Plotting temperature data
figure;
subplot(3,1,1);
plot(1:num_samples, temperature_data, 'b.-');
xlabel('Time (samples)');
ylabel('Temperature (Celsius)');
title('Temperature Variation');
 
% Plotting humidity data
subplot(3,1,2);
plot(1:num_samples, humidity_data, 'g.-');
xlabel('Time (samples)');
ylabel('Humidity (%)');
title('Humidity Variation');
 
% Plotting rainfall data
subplot(3,1,3);
bar(1:num_samples, rainfall_data, 'r');
xlabel('Time (samples)');
ylabel('Rainfall (mm)');
title('Rainfall Variation');
 
subplot(3,1,3)
colorbar
 
% Statistical analysis
mean_temperature = mean(temperature_data);
mean_humidity = mean(humidity_data);
total_rainfall = sum(rainfall_data);
 
disp(['Mean Temperature: ', num2str(mean_temperature), ' Celsius']);
Mean Temperature: 28.01 Celsius
disp(['Mean Humidity: ', num2str(mean_humidity), ' %']);
Mean Humidity: 58.62 %
disp(['Total Rainfall: ', num2str(total_rainfall), ' mm']);
Total Rainfall: 2523 mm
